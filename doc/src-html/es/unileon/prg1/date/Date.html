037
038                if (isMonthRight(month)) {
039
040                        this.month = month;
041                }else{
042
043                        throw new DateException("Mes "+month+" no valido, debe ser un mes entre 1 y 12");
044                }
045
046                if (isDayofMonthRight(day,month)) {
047
048                        this.day = day;
049                }else{
050
051                        throw new DateException("Fecha no valida: (" + day + "/" + month + "/" + year + ")");
052                }
053                if (year > 0) {
054                        this.year = year;                       
055                }else{
056                        throw new DateException("Anyo no valido");
057                }
058        }
