014import java.io.* ;
015/**
016 *  Clase que permite leer valores simples de la linea de comandos.
017 *  Si ocurre un error durante la lectura las excepciones lanzadas 
018 *  son capturadas y  se retorna el menor valor representable
019 *  del tipo de datos leido.
020 *  
021 *  Nota: esta no es una muy buena opcion de tratar los posibles errores. En futuras versiones 
022 *  (cuendo se explique la gestion de excepciones) se mejorara este tratamiento.
023 *
024 *  @version 1.0
025 *  @author PRG
026 */
027public class Teclado {
028  
029  /**
030  * Stream para leer lineas de la entrada standar (por defecto el teclado) 
031  * 
032  * @see BufferedReader
033  * @see InputStreamReader
034  * @see System.in
035  */
036  private static final BufferedReader in =
037    new BufferedReader(new InputStreamReader (System.in)) ;
038  /**
039   * Atributo para almacenar la linea ledia por teclado
040   * 
041   * @see String
042   */
043  private static String input;
044  
045  /**
046   *  Lee un <CODE>int</CODE> introducido por el teclado. 
047   *  
048   *  
049   *  @return Retorna el valor introducido por el teclado. Si se introduce un valor que no es correcto se retorna el minimo
050   *  valor representable  en int (Integer.MIN_VALUE)
051   *  @see Integer
052   */
053  public static final synchronized int readInteger() {
054    input = "";
055    int value = Integer.MIN_VALUE;
056    try {
057      input = in.readLine() ;
058    }
059    catch (IOException e) { error("standard input"); } 
060    try {
061      value = Integer.parseInt(input) ;
062    }
063    catch (NumberFormatException e) { 
064                value=Integer.MIN_VALUE;
065    }
066    return value ;
067  }
068
069  /**
070   *  Lee un <CODE>short</CODE> introducido por el teclado. 
071   *  
072   *  
073   *  @return Retorna el valor introducido por el teclado. Si se introduce un valor que no es correcto se retorna el minimo
074   *  valor representable  en short (Short.MIN_VALUE)
075   *  @see Short
076   */
077  public static final synchronized short readShort(){
078    input = "";
079    short value = Short.MIN_VALUE;
080    try {
081      input = in.readLine() ;
082    }
083    catch (IOException e) { error("standard input"); } 
084    try {
085      value = Short.parseShort(input) ;
086    }
087    catch (NumberFormatException e) {
088        value = Short.MIN_VALUE;        
089    }
090    return value ;
091    
092  }
093  
094  /**
095   *  Lee un <CODE>long</CODE> introducido por el teclado. 
096   *  
097   *  
098   *  @return Retorna el valor introducido por el teclado. Si se introduce un valor que no es correcto se retorna el minimo
099   *  valor representable  en long (Long.MIN_VALUE)
100   *  @see Long
101   */
102  
103  public static final synchronized long readLong() {
104    input = "" ;
105    long value = Long.MIN_VALUE;
106    try {
107      input = in.readLine() ;
108    }
109    catch (IOException e) { error("standard input"); }
110    try {
111      value = Long.parseLong(input) ;
112    }
113    catch (NumberFormatException e) {
114        value = Long.MIN_VALUE; 
115    }
116    return value ;
117  }
118  
119  
120  /**
121   *  Lee un <CODE>float</CODE> introducido por el teclado. 
122   *  
123   *  
124   *  @return Retorna el valor introducido por el teclado. Si se introduce un valor que no es correcto se retorna el minimo
125   *  valor representable  en float (Float.MIN_VALUE)
126   *  @see Float
127   *    
128   */
129  public static final synchronized float readFloat() {
130    input = "" ;
131    float value = Float.MIN_VALUE ;
132    try {
133      input = in.readLine() ;
134    }
135    catch (IOException e) { error("standard input"); }
136    try {
137      value = Float.parseFloat(input) ;
138    }
139    catch (NumberFormatException e) {
140        value = Float.MIN_VALUE ;
141    }
142    return value ;
143  }
144  
145  
146  /**
147   *  Lee un <CODE>long</CODE> introducido por el teclado. 
148   *  
149   *  @return Retorna el valor introducido por el teclado. Si se introduce un valor que no es correcto se retorna el minimo
150   *  valor representable  en double (Double.MIN_VALUE)
151   *  @see Double
152   */
153  
154  public static final synchronized double readDouble() {
155    input = "" ;
156    double value = Double.MIN_VALUE ;
157    try {
158      input = in.readLine() ;
159    }
160    catch (IOException e) { error("standard input"); }
161    try {
162      value = Double.parseDouble(input) ;
163    }
164    catch (NumberFormatException e) { 
165        value = Double.MIN_VALUE;
166    }
167    return value ;
168  }
169  
170  
171  
172  
173  /**
174   *  Lee un <CODE>char</CODE> introducido por el teclado. 
175   *  
176   *  @return Retorna el caracter introducido por el teclado.
177   */
178  public static final synchronized char readChar() {
179    char c = ' ' ;
180    try {
181     c = (char)in.read() ;     
182    }
183    catch (IOException e) { error("standard input"); }
184    return c ;
185  }
186  
187  /**
188   *  Lee un <CODE>String</CODE> introducido por el teclado.
189   *  
190   *  @return Retorna la cadena de caracteres introducida por el teclado.
191   *  @see String
192   */
193  public static final synchronized String readString() {
194    String s = "";
195    try {
196      s = in.readLine() ;
197    }
198    catch (IOException e) { error("standard input"); }
199    if (s == null) {
200      s = "" ;
201    }
202    return s ;
203  }
204  
205  /**
206   * Lanzado cuando ocurre un error.Muestra un mensaje y termina
207   */
208  private static void error(String msg) {
209    System.err.println("Error: " + msg) ;
210    System.err.println("Imposible continuar la ejecucion del programa") ;
211    System.exit(0) ;
212  }
213}
